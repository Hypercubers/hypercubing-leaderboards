#[derive(Debug, Default, Copy, Clone, PartialEq, Eq, Hash)]
pub enum AuthRequestStatus {
    #[default]
    Waiting,
    Confirmed,
    Expired,
}
impl IntoResponse for AuthRequestStatus {
    fn into_response(self) -> axum::response::Response {
        let status_code = match self {
            AuthRequestStatus::Waiting => StatusCode::PROCESSING,
            AuthRequestStatus::Confirmed => StatusCode::ACCEPTED,
            AuthRequestStatus::Expired => StatusCode::BAD_REQUEST,
        };
        (status_code, axum::response::NoContent).into_response()
    }
}
